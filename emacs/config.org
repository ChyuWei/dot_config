  ;;; defines.el --- defines -*- lexical-binding: t -*-


  ;;; Code:

  ;; system define
  (defconst os/linuxp
    (eq system-type 'gnu/linux))

  (defconst os/macp
    (eq system-type 'darwin))

  ;; config org file
  (defconst priconf/conf
    (concat user-emacs-directory "config.org"))

  (defun priconf/open ()
    "Edit config.org"
    (interactive)
    (find-file priconf/conf))

  (defun priconf/tangle ()
    "Org tangle config.org"
    (interactive)
    (org-babel-tangle-file priconf/conf))

  ;; transparent
  (defun transparent-change (fn)
    (let* ((alpha (or (frame-parameter nil 'alpha) 100))
           (new-alpha (funcall fn alpha)))
      (progn (set-frame-parameter nil 'alpha new-alpha)
             (message "current alpha %d" new-alpha))))

  (defun transparent-add ()
    (interactive)
    (transparent-change (lambda (x) (min 100 (+ x 10)))))

  (defun transparent-sub ()
    (interactive)
    (transparent-change (lambda (x) (max 0 (- x 10)))))

  ;; proxy function
  (defun proxy-socks-show ()
    "Show SOCKS proxy."
    (interactive)
    (if (bound-and-true-p socks-noproxy)
        (message "Current SOCKS%d proxy is %s:%s"
                 (cadddr socks-server) (cadr socks-server) (caddr socks-server))
      (message "No SOCKS proxy")))

  (defun proxy-socks-enable ()
    "Enable socks proxy"
    (interactive)
    (require 'socks)
    (setq url-gateway-method 'socks
          socks-noproxy '("localhost")
          socks-server `("Default server" "127.0.0.1" 1080 5))
    (setenv "all_proxy" "socks5://127.0.0.1:1080")
    (proxy-socks-show))

  (defun proxy-socks-disable ()
    "Disable SOCKS proxy."
    (interactive)
    (setq url-gateway-method 'native
          socks-noproxy nil
          socks-server nil)
    (setenv "all_proxy" "")
    (proxy-socks-show))

  ;; ui function
  (defun font-installed-p (font-name)
    (find-font (font-spec :name font-name)))

  (defvar after-load-theme-hook nil)
  (defun run-after-load-theme-hook (&rest _)
    (run-hooks 'after-load-theme-hook))
  (advice-add #'load-theme :after #'run-after-load-theme-hook)

  ;; helper function
  (defun insert-timestamp ()
    (interactive)
    (insert (format-time-string "%Y-%m-%dT%H:%M")))

  (provide 'defines)

  ;;; defines.el ends here

        self/backup-theme 'doom-dark+

  (defun self/toggle-theme ()
    (interactive)
    (when (display-graphic-p)
      (progn (if self/is-default-theme
                 (load-theme self/default-theme t)
               (load-theme self/backup-theme t))
             (setq self/is-default-theme (not self/is-default-theme)))))

  (self/toggle-theme)

  (use-package doom-modeline
    :hook (after-init . doom-modeline-mode))

  ;; font
  (setq my-default-font-list '("Cascadia Code" "Fira Code" "Jetbrains Mono" "SF Mono" "Hack" "Source Code Pro" "Menlo"
                               "Monaco" "DejaVu Sans Mono" "Consolas")
        my-mode-line-font-list '("Menlo" "SF Pro Display" "Helvetica")
        my-unicode-font-list '("Segoe UI Symbol" "Symbola" "Symbol")
        my-emoji-font-list '("Noto Color Emoji" "Apple Color Emoji" "Segoe UI Emoji")
        my-chinese-font-list '("LXGW WenKai Screen" "WenQuanYi Micro Hei Mono" "PingFang SC" "Microsoft Yahei UI"
                               "Simhei"))


  (when (display-graphic-p)
    ;; Set default font
    (cl-loop for font in my-default-font-list
             when (font-installed-p font)
             return (set-face-attribute 'default nil
                                        :family font
                                        :height 150))

    ;; Set mode-line font
    (cl-loop for font in my-mode-line-font-list
             when (font-installed-p font)
             return (progn
                      (set-face-attribute 'mode-line nil :family font :height 120)

    ;; Emoji

             when (font-installed-p font)
             return (cond
                     ((< emacs-major-version 27)
                      (set-fontset-font "fontset-default" 'unicode font nil 'prepend))
                     ((< emacs-major-version 28)
                      (set-fontset-font t 'symbol (font-spec :family font) nil 'prepend))
                     (t
                      (set-fontset-font t 'emoji (font-spec :family font) nil 'prepend))))

    ;; Specify font for Chinese characters
    (cl-loop for font in my-chinese-font-list
             when (font-installed-p font)
             return (progn
                      (setq face-font-rescale-alist `((,font . 1.0)))
                      (set-fontset-font t 'han (font-spec :family font)))))

  (add-hook 'after-init-hook 'global-hl-line-mode)

                    node-end (treesit-node-end node))))
          (set-mark node-end)
          (goto-char node-start)))
      (add-to-list 'er/try-expand-list 'treesit-mark-bigger-node))
